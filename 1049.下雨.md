1049.下雨这是一个**二分查找 + 分数近似**的经典算法题。

---

## 一、问题背景

### 题目场景

有一个**宽度为 W 的水槽**，每个位置 `i` 的高度为 `H[i]`。向水槽中倒入 **V 体积**的水，水会填满低洼处，最终水面会达到一个**统一高度 L**。

### 物理原理

```
水槽剖面图：
    |  H[0]  |  H[1]  |  H[2]  |
    |  5     |  3     |  4     |
    |________|________|________|
    
倒入水后，水面会趋于水平：
    |~~~~~~~~|~~~~~~~~|~~~~~~~~|  ← 水面高度 L
    |  5  ██  |  3  ██  |  4  ██  |
    |________|________|________|
```

**水量计算公式**：
```
总水量 V = Σ max(0, L - H[i])  (当 L > H[i] 时，该位置被水覆盖)
```

## 三、逐段详解

### 1. 最大公约数函数

```cpp
long long gcd_val(long long a, long long b) {
    return b == 0 ? a : gcd_val(b, a % b);  // 欧几里得算法
}
```

**作用**：用于将分数 `n/d` 化简为最简形式。

**示例**：
```cpp
gcd_val(6, 9) = 3  // 6/9 → 2/3
gcd_val(10, 25) = 5  // 10/25 → 2/5
```

---

### 2. 输入读取

```cpp
long long W, V;  // W = 水槽宽度（位置数），V = 倒入的水量
if (!(cin >> W >> V)) return 0;  // 如果读取失败，直接退出

vector<long long> H(W);  // 存储每个位置的高度
long long min_H = 2e18;  // 初始化为极大值（2 × 10^18）
for (int i = 0; i < W; ++i) {
    cin >> H[i];
    min_H = min(min_H, H[i]);  // 记录最小高度
}
```

---

### 3. 二分查找水面高度（核心）

```cpp
double low = min_H;           // 水面最低可能高度 = 最低点高度
double high = min_H + V + 1.0; // 水面最高可能高度（足够高，保证能装下所有水）

for (int iter = 0; iter < 100; ++iter) {  // 迭代100次，精度足够
    double mid = low + (high - low) / 2.0;  // 中间值
    
    double current_water = 0;  // 计算在高度 mid 时能装多少水
    for (int i = 0; i < W; ++i) {
        if (mid > H[i]) {
            current_water += (mid - H[i]);  // 该位置能容纳的水量
        }
    }
    
    if (current_water < V) {
        low = mid;   // 水不够，说明水面太低，需要升高
    } else {
        high = mid;  // 水太多或刚好，说明水面太高，需要降低
    }
}
double final_L = low;  // 最终近似水面高度
```

#### 二分查找详细示例

```
输入：W=3, V=10, H=[5, 3, 4], min_H=3

迭代过程：
初始：low=3, high=14 (3+10+1)

iter 0: mid=8.5, water=(8.5-5)+(8.5-3)+(8.5-4)=3.5+5.5+4.5=13.5 > V
        → high=8.5 (水面太高，降低)

iter 1: mid=5.75, water=(5.75-5)+(5.75-3)+(5.75-4)=0.75+2.75+1.75=5.25 < V
        → low=5.75 (水面太低，升高)

iter 2: mid=7.125, water=(7.125-5)+(7.125-3)+(7.125-4)=2.125+4.125+3.125=9.375 < V
        → low=7.125 (继续升高)

iter 3: mid=7.8125, water=(7.8125-5)+(7.8125-3)+(7.8125-4)=2.8125+4.8125+3.8125=11.4375 > V
        → high=7.8125 (降低)

... 经过100次迭代，最终 final_L ≈ 7.5

验证：在 L=7.5 时，water=(7.5-5)+(7.5-3)+(7.5-4)=2.5+4.5+3.5=10.5 ≈ V（实际是二分逼近）
```

---

### 4. 分数近似（将浮点数转为分数）

```cpp
long long best_num = 0;    // 最佳分子
long long best_den = 1;    // 最佳分母
double min_diff = 1e9;     // 最小误差（初始化为极大值）

// 遍历所有可能的分母 d = 1 到 W
for (long long d = 1; d <= W; ++d) {
    long long n = round(final_L * d);  // 四舍五入得到分子
    double val = (double)n / d;        // 计算当前分数的值
    
    // 如果当前分数更接近 final_L，则更新
    if (abs(val - final_L) < min_diff) {
        min_diff = abs(val - final_L);
        best_num = n;
        best_den = d;
    }
}
```

#### 为什么分母范围是 `1` 到 `W`？

题目要求输出**最简分数**，分母的最大值通常限制为 `W`（水槽宽度），这是为了确保分数的精度和合理性。

#### 分数近似示例

```
final_L = 7.5

d=1: n=round(7.5)=8, val=8.0, diff=0.5
d=2: n=round(15)=15, val=7.5, diff=0.0 ✅ 最佳
d=3: n=round(22.5)=23, val=7.667, diff=0.167
...

最佳：15/2 = 7.5
```

---

### 5. 分数化简

```cpp
long long g = gcd_val(best_num, best_den);  // 计算最大公约数
best_num /= g;   // 分子化简
best_den /= g;   // 分母化简
```

**示例**：
```cpp
best_num = 15, best_den = 6
g = gcd_val(15, 6) = 3
best_num = 15/3 = 5
best_den = 6/3 = 2
结果：5/2
```

---

### 6. 分数输出格式化

```cpp
if (best_den == 1) {
    cout << best_num << endl;  // 整数：直接输出
} else {
    long long int_part = best_num / best_den;     // 整数部分
    long long rem_num = best_num % best_den;      // 余数部分（分子）
    
    if (rem_num == 0) {
        cout << int_part << endl;  // 整数：直接输出
    } else if (int_part == 0) {
        cout << rem_num << "/" << best_den << endl;  // 真分数：a/b
    } else {
        cout << int_part << "+" << rem_num << "/" << best_den << endl;  // 带分数：a+b/c
    }
}
```

#### 输出格式对照表

| 分数 | 化简后 | int_part | rem_num | 输出 |
| :--- | :--- | :--- | :--- | :--- |
| 8/1 | 8 | 8 | 0 | `8` |
| 3/2 | 3/2 | 1 | 1 | `1+1/2` |
| 2/3 | 2/3 | 0 | 2 | `2/3` |
| 15/6 | 5/2 | 2 | 1 | `2+1/2` |

| 步骤 | 时间复杂度 | 说明 |

## 七、可能的测试点错误

### ❌ 错误1：二分迭代次数不够
```cpp
for (int iter = 0; iter < 100; ++iter)  // 100次通常足够
// 如果精度要求高，可以改为 200 次
```

### ❌ 错误2：`low` 和 `high` 初始值不对
```cpp
double low = min_H;           // ✅ 正确：水面不会低于最低点
double high = min_H + V + 1.0; // ✅ 正确：足够高
```

### ❌ 错误3：分数近似时 `d` 的范围
```cpp
for (long long d = 1; d <= W; ++d)  // ✅ 通常为 1 到 W
// 如果是其他范围（如 1 到 1000），可能需要调整
```

### ❌ 错误4：输出格式中的 `+` 号
```cpp
cout << int_part << "+" << rem_num << "/" << best_den;  // 带分数格式
// 有些题目要求输出 "a b/c"（空格），而不是 "a+b/c"
```
好的，我来给代码使用更简洁的变量命名（通常是算法竞赛风格的短命名）：

```cpp
#include <bits/stdc++.h>
using namespace std;

long long gcd_val(long long a, long long b) {
    return b == 0 ? a : gcd_val(b, a % b);  // 欧几里得算法：求最大公约数
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long W, V;                         // W=宽度，V=水量
    if (!(cin >> W >> V)) return 0;
    
    vector<long long> H(W);                 // H=高度数组
    long long mn = 2e18;                    // mn=最小高度
    for (int i = 0; i < W; ++i) {
        cin >> H[i];
        mn = min(mn, H[i]);
    }
    
    double l = mn;                          // l=下界
    double r = mn + V + 1.0;                // r=上界
    
    for (int it = 0; it < 100; ++it) {      // it=迭代次数
        double m = l + (r - l) / 2.0;       // m=中间值（候选水面高度）
        double water = 0;                   // water=当前水量
        for (int i = 0; i < W; ++i) {
            if (m > H[i]) water += (m - H[i]);
        }
        if (water < V) l = m;               // 水量不足，升高水面
        else r = m;                         // 水量过多，降低水面
    }
    double L = l;                           // L=最终水面高度
    
    long long num = 0;                      // num=分子
    long long den = 1;                      // den=分母
    double eps = 1e9;                       // eps=最小误差
    
    for (long long d = 1; d <= W; ++d) {    // d=分母
        long long n = round(L * d);         // n=分子
        double val = (double)n / d;         // val=分数值
        if (abs(val - L) < eps) {
            eps = abs(val - L);
            num = n;
            den = d;
        }
    }
    
    long long g = gcd_val(num, den);        // g=最大公约数
    num /= g;
    den /= g;
    
    if (den == 1) {
        cout << num << endl;
    } else {
        long long ip = num / den;           // ip=整数部分
        long long rm = num % den;           // rm=余数（新分子）
        if (rm == 0) {
            cout << ip << endl;
        } else if (ip == 0) {
            cout << rm << "/" << den << endl;
        } else {
            cout << ip << "+" << rm << "/" << den << endl;
        }
    }
    
    return 0;
}
```

## 更极简版本（适合代码竞赛）

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll gcd(ll a, ll b) { return b ? gcd(b, a % b) : a; }

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    ll W, V;
    cin >> W >> V;
    vector<ll> H(W);
    ll mn = 4e18;
    for (int i = 0; i < W; ++i) cin >> H[i], mn = min(mn, H[i]);
    
    double l = mn, r = mn + V + 1;
    for (int it = 0; it < 100; ++it) {
        double m = (l + r) / 2, w = 0;
        for (int i = 0; i < W; ++i)
            if (m > H[i]) w += m - H[i];
        w < V ? l = m : r = m;
    }
    
    ll num = 0, den = 1;
    double eps = 1e9;
    for (ll d = 1; d <= W; ++d) {
        ll n = round(l * d);
        double v = 1.0 * n / d;
        if (abs(v - l) < eps) eps = abs(v - l), num = n, den = d;
    }
    
    ll g = gcd(num, den);
    num /= g, den /= g;
    
    if (den == 1) cout << num << '\n';
    else {
        ll ip = num / den, rm = num % den;
        if (rm == 0) cout << ip << '\n';
        else if (ip == 0) cout << rm << '/' << den << '\n';
        else cout << ip << '+' << rm << '/' << den << '\n';
    }
}
```
