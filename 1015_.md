## 完整代码（先贴完整版）

```cpp
#include <bits/stdc++.h>
using namespace std;

bool isGE(string a, string b) {
    int da = a.find('.');
    int db = b.find('.');
    string ai = (da == -1) ? a : a.substr(0, da);
    string af = (da == -1) ? "" : a.substr(da + 1);
    string bi = (db == -1) ? b : b.substr(0, db);
    string bf = (db == -1) ? "" : b.substr(db + 1);
    if (ai.empty()) ai = "0";
    if (bi.empty()) bi = "0";
    if (ai.size() != bi.size()) return ai.size() > bi.size();
    if (ai != bi) return ai > bi;
    int len = max(af.size(), bf.size());
    while ((int)af.size() < len) af += "0";
    while ((int)bf.size() < len) bf += "0";
    return af >= bf;
}

string addOne(string s) {
    int c = 1;
    for (int i = (int)s.size() - 1; i >= 0 && c; i--) {
        int v = (s[i] - '0') + c;
        s[i] = char(v % 10 + '0');
        c = v / 10;
    }
    if (c) s = "1" + s;
    return s;
}

string sub(string a, string b, int n) {
    int da = a.find('.');
    int db = b.find('.');
    string ai = (da == -1) ? a : a.substr(0, da);
    string af = (da == -1) ? "" : a.substr(da + 1);
    string bi = (db == -1) ? b : b.substr(0, db);
    string bf = (db == -1) ? "" : b.substr(db + 1);
    if (ai.empty()) ai = "0";
    if (bi.empty()) bi = "0";

    int flen = max(max((int)af.size(), (int)bf.size()), n + 1);
    while ((int)af.size() < flen) af += "0";
    while ((int)bf.size() < flen) bf += "0";

    int ilen = max((int)ai.size(), (int)bi.size());
    while ((int)ai.size() < ilen) ai = "0" + ai;
    while ((int)bi.size() < ilen) bi = "0" + bi;

    string fsum = "";
    int borrow = 0;
    for (int i = flen - 1; i >= 0; i--) {
        int x = (af[i] - '0') - borrow;
        int y = (bf[i] - '0');
        if (x < y) {
            x += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        fsum = char(x - y + '0') + fsum;
    }

    string isum = "";
    for (int i = ilen - 1; i >= 0; i--) {
        int x = (ai[i] - '0') - borrow;
        int y = (bi[i] - '0');
        if (x < y) {
            x += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        isum = char(x - y + '0') + isum;
    }

    if ((int)fsum.size() > n) {
        if (fsum[n] >= '5') {
            string tmp = fsum.substr(0, n);
            int c = 1;
            for (int i = n - 1; i >= 0 && c; i--) {
                int v = (tmp[i] - '0') + c;
                tmp[i] = char(v % 10 + '0');
                c = v / 10;
            }
            if (c) isum = addOne(isum);
            fsum = tmp;
        } else {
            fsum = fsum.substr(0, n);
        }
    }
    while ((int)fsum.size() < n) fsum += "0";

    int p = 0;
    while (p + 1 < (int)isum.size() && isum[p] == '0') p++;
    isum = isum.substr(p);

    return isum + "." + fsum;
}

int main() {
    string a, b;
    int n;
    cin >> a >> b >> n;

    bool neg = false;
    if (!isGE(a, b)) {
        neg = true;
        swap(a, b);
    }

    string ans = sub(a, b, n);
    if (neg) cout << "-";
    cout << ans << endl;

    return 0;
}
```

### 第三部分：`addOne` 函数（大整数 +1）

```cpp
string addOne(string s) {
    int c = 1;
    for (int i = (int)s.size() - 1; i >= 0 && c; i--) {
        int v = (s[i] - '0') + c;
        s[i] = char(v % 10 + '0');
        c = v / 10;
    }
    if (c) s = "1" + s;
    return s;
}
```

**作用：** 将字符串表示的大整数加 1

**示例：**

| 输入 | 输出 |
|------|------|
| `"0"` | `"1"` |
| `"9"` | `"10"` |
| `"99"` | `"100"` |
| `"123"` | `"124"` |

**逐行解释：**

| 行 | 含义 |
|----|------|
| `int c = 1` | 要加的数（1） |
| `for (int i = size-1; i >= 0 && c; i--)` | 从最低位开始，只要有进位就继续 |
| `int v = (s[i] - '0') + c` | 当前位 + 进位 |
| `s[i] = char(v % 10 + '0')` | 当前位结果 |
| `c = v / 10` | 新的进位 |
| `if (c) s = "1" + s` | 最高位还有进位，前面补 `1` |

### 第2步：小数部分补 `0`

```cpp

**作用：** 小数部分右侧补 `0` 到相同长度

- 补 `0` 到 `flen` 位

**示例：**

```

### 第4步：小数部分相减（核心借位逻辑）

```cpp
    string fsum = "";
    int borrow = 0;
    for (int i = flen - 1; i >= 0; i--) {
        int x = (af[i] - '0') - borrow;
        int y = (bf[i] - '0');
        if (x < y) {
            x += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        fsum = char(x - y + '0') + fsum;
    }
```

**逐行解释：**

| 行 | 含义 |
|----|------|
| `string fsum = ""` | 存储小数部分结果 |
| `int borrow = 0` | 借位，初始为 0 |
| `for (int i = flen - 1; i >= 0; i--)` | 从**最低位**（最右边）开始遍历 |
| `int x = (af[i] - '0') - borrow` | 被减数当前位 - 借位 |
| `int y = (bf[i] - '0')` | 减数当前位 |
| `if (x < y)` | 如果被减数 < 减数，需要借位 |
| `x += 10` | 被减数加 10 |
| `borrow = 1` | 标记借位 |
| `else { borrow = 0 }` | 不需要借位 |
| `fsum = char(x - y + '0') + fsum` | 结果位插入到最前面 |

**借位逻辑图示：**
```
  3 2 1
-   4 5
───────
  2 7 6

个位：1 < 5 → 1+10=11, borrow=1, diff=11-5=6
十位：2 - 1 = 1 < 4 → 1+10=11, borrow=1, diff=11-4=7
百位：3 - 1 = 2 < 0 → 2 >= 0, borrow=0, diff=2-0=2
```

### 第6步：四舍五入

```cpp
    if ((int)fsum.size() > n) {
        if (fsum[n] >= '5') {
            string tmp = fsum.substr(0, n);
            int c = 1;
            for (int i = n - 1; i >= 0 && c; i--) {
                int v = (tmp[i] - '0') + c;
                tmp[i] = char(v % 10 + '0');
                c = v / 10;
            }
            if (c) isum = addOne(isum);
            fsum = tmp;
        } else {
            fsum = fsum.substr(0, n);
        }
    }
```

**四舍五入逻辑：**

1. 检查第 `n` 位（要舍弃的第一位）
2. 如果 `< '5'`，直接截断
3. 如果 `>= '5'`：
   - 取前 `n` 位
   - 从第 `n-1` 位开始加 1（进位）
   - 如果进位溢出，整数部分加 1

## 总结

| 步骤 | 关键点 |
|------|--------|
| **大小比较** | `isGE` 函数，先比整数再比小数 |
| **借位处理** | `x - borrow < y` 时借 10 |
| **四舍五入** | 检查第 n 位，>=5 进位 |
| **负数处理** | 主函数中判断并标记 |

**核心思想：用字符串模拟竖式减法，从最低位开始逐位相减并处理借位。**
