## 三、`trie.h` - Trie类声明

```cpp
/**
 * trie.h - Trie字典树类声明
 * 作用：定义数据结构和方法接口，类似于"目录"
 */

#ifndef TRIE_H
#define TRIE_H

#include <string>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <memory>
#include <utility>

using namespace std;

// ============================================================
// 1. TrieNode 结构体 - 字典树的节点
// ============================================================
// 
// 什么是 Trie（字典树）？
// ─────────────────────────────────────────────────────────────
// Trie 是一种树形数据结构，用于高效存储和查找字符串。
// 每个节点代表一个字符，从根到某个节点的路径形成一个前缀。
//
// 示例：存储 and, ando, another
//
//          root
//         /    \
//        a      t
//       / \      \
//      n   n      a
//     / \   \      \
//    d   o   o      n
//   /     \   \      \
//  and    ando anot   d
//  (6)    (4)  (3)    |
//                    tand
//                    (7)
//
// 每个节点存储：
//   - children: 子节点（当前字符的下一个字符）
//   - isWord:   是否是一个单词的结尾
//   - freq:     词频（仅当 isWord=true 时有效）
//   - word:     完整单词（仅当 isWord=true 时有效）
struct TrieNode {
    // 子节点映射
    // key: 字符 ('a' ~ 'z')
    // value: 指向子节点的智能指针
    // 
    // 为什么用 unique_ptr？
    // → 自动管理内存，不需要手动 delete
    // → 每个节点只能有一个父节点，所有权唯一
    // → 安全，不会内存泄漏
    unordered_map<char, unique_ptr<TrieNode>> children;

    // 是否是一个单词的结尾节点
    // true: 从根到该节点的路径构成一个完整单词
    // false: 只是一个前缀，不是完整单词
    bool isWord = false;

    // 词频：该单词被用户选择的累计次数
    // 仅当 isWord == true 时有效
    // 用于排序：词频越高越靠前
    int freq = 0;

    // 完整单词字符串
    // 仅当 isWord == true 时有效
    // 存储完整单词，方便直接读取
    string word;
};

// ============================================================
// 2. Trie 类 - 字典树的核心操作
// ============================================================
//
// 职责：
//   - 存储所有单词及其词频
//   - 提供添加、删除、更新词频功能
//   - 提供数字查询、前缀查询、通配符查询
//   - 管理 Top-K 缓存
class Trie {
public:
    // ---------- 构造函数 ----------
    // 初始化：创建根节点
    Trie();

    // ---------- 词库维护（增删改）----------
    // 添加单词
    // @param word: 要添加的单词（小写字母）
    // @param freq: 初始词频（正整数）
    // @return true: 添加成功，false: 失败（已存在或非法）
    bool addWord(const string& word, int freq);

    // 删除单词
    // @param word: 要删除的单词
    // @return true: 删除成功，false: 单词不存在
    bool deleteWord(const string& word);

    // 更新词频
    // @param word: 要更新的单词
    // @param freq: 新的词频（正整数）
    // @return true: 更新成功，false: 单词不存在
    bool updateFreq(const string& word, int freq);

    // ---------- 查询功能 ----------
    // 精确查询词频
    // @param word: 要查询的单词
    // @return 词频值，如果不存在返回 -1
    int getFreq(const string& word) const;

    // 全词库导出
    // @return 所有单词及词频，按字典序排序
    vector<pair<string, int>> getAllWords() const;

    // 前缀查询
    // @param prefix: 要查询的前缀
    // @return 所有以 prefix 开头的单词，按"词频降序+字典序升序"
    vector<pair<string, int>> prefixQuery(const string& prefix) const;

    // 数字序列查询（核心功能！）
    // @param digits: 数字序列（如 "26"）
    // @return 匹配该数字序列的 Top-K 个单词
    // 
    // 数字匹配规则：
    // 2→abc, 3→def, 4→ghi, 5→jkl,
    // 6→mno, 7→pqrs, 8→tuv, 9→wxyz
    vector<pair<string, int>> digitQuery(const string& digits) const;

    // 通配符匹配
    // @param pattern: 匹配模式
    //   - '?' 匹配任意单个字符
    //   - '*' 匹配任意长度字符串（包括空串）
    // @return 所有匹配的单词，按字典序排序
    vector<string> matchPattern(const string& pattern) const;

    // ---------- 清空 ----------
    // 清空所有数据
    void clear();

    // ---------- 缓存管理 ----------
    // 清空所有缓存
    void invalidateAllCache();

    // 清空指定数字序列的缓存
    void invalidateCacheByDigits(const string& digits);

    // 清空单词相关的所有缓存（影响所有前缀）
    void invalidateCacheByWord(const string& word);

    // 清空前缀相关的缓存
    void invalidateCacheByPrefix(const string& prefix);

    // 打印缓存统计（调试用）
    void printCacheStats() const;

private:
    // ============================================================
    // 3. 私有成员变量
    // ============================================================

    // 根节点（树的入口）
    // unique_ptr 自动管理根节点的生命周期
    unique_ptr<TrieNode> root_;

    // 单词集合（快速判断单词是否存在）
    // 哈希集合，查找 O(1)
    // 用于 add/delete/update 时快速检查单词是否存在
    unordered_set<string> wordSet_;

    // 词频映射表（快速查询词频）
    // 哈希表，查找 O(1)
    // 用于 getFreq 时快速返回结果
    // 也用于 getAllWords 时遍历所有单词
    unordered_map<string, int> wordFreq_;

    // ===== Top-K 缓存 =====
    // key: 数字序列 (如 "26")
    // value: 该数字序列对应的 Top-K 候选列表 (已排序)
    // 
    // 为什么要缓存？
    // → 用户可能多次输入相同的数字序列
    // → 每次从 Trie 查询都要遍历和排序，耗时
    // → 缓存可以 O(1) 返回结果
    //
    // 什么时候更新缓存？
    // → 任何词库修改（add/delete/update）都需要清空相关缓存
    // → 选择单词后词频变化，也需要清空
    unordered_map<string, vector<pair<string, int>>> cache_;

    // ============================================================
    // 4. 私有辅助函数
    // ============================================================

    // 校验单词是否合法（非空、全小写字母）
    bool isValidWord(const string& word) const;

    // 根据前缀查找对应的节点
    // @param prefix: 要查找的前缀
    // @return 指向该节点的指针，不存在返回 nullptr
    TrieNode* findNode(const string& prefix) const;

    // DFS 收集从当前节点开始的所有完整单词
    // @param node: 当前节点
    // @param result: 收集结果的引用
    void collectWords(TrieNode* node, vector<pair<string, int>>& result) const;

    // 数字匹配的递归实现
    // @param node: 当前节点
    // @param current: 当前已匹配的字符串
    // @param digits: 完整的数字序列
    // @param pos: 当前处理的数字位置
    // @param result: 收集结果的引用
    void collectByDigit(TrieNode* node, const string& current,
                        const string& digits, size_t pos,
                        vector<pair<string, int>>& result) const;

    // 通配符匹配的递归实现
    void collectByPattern(TrieNode* node, const string& current,
                          const string& pattern, size_t pos,
                          vector<string>& result) const;

    // 排序：词频降序，字典序升序
    // 任务书要求的统一排序规则
    static void sortByFreqAndLex(vector<pair<string, int>>& items);

    // ===== 缓存辅助函数 =====
    // 单词转数字序列
    // 例: "hello" → "43556"
    string wordToDigits(const string& word) const;

    // 前缀转数字序列
    // 例: "hel" → "435"
    string prefixToDigits(const string& prefix) const;

    // 收集单词的所有前缀对应的数字序列
    // 例: "hello" → ["4", "43", "435", "4355", "43556"]
    void collectAllPrefixDigits(const string& word,
                                vector<string>& prefixes) const;

    // 从前缀开始收集所有前缀对应的数字序列
    void collectAllPrefixDigitsFromPrefix(const string& prefix,
                                          vector<string>& prefixes) const;
};

#endif
```

```cpp
string getWord(node) {
    string result;
    while (node->parent) {
        result = node->ch + result;  // 从后往前拼
        node = node->parent;
```

## 四、词库维护函数详解

### 1. addWord - 添加单词

```cpp
bool Trie::addWord(const string& word, int freq) {
    // ----- 步骤1：参数校验 -----
    if (!isValidWord(word) || freq <= 0 || wordSet_.count(word))
        return false;

    // ----- 步骤2：遍历创建路径 -----
    TrieNode* node = root_.get();
    for (char ch : word) {
        if (!node->children.count(ch))
            node->children[ch] = make_unique<TrieNode>();  // 创建新节点
        node = node->children[ch].get();                   // 移动到子节点
    }
    
    // ----- 步骤3：设置单词信息 -----
    node->isWord = true;
    node->freq = freq;
    node->word = word;
    
    // ----- 步骤4：更新辅助索引 -----
    wordSet_.insert(word);
    wordFreq_[word] = freq;

    // ----- 步骤5：缓存失效 -----
    invalidateCacheByString(word);

    return true;
}
```

### 2. deleteWord - 删除单词

```cpp
bool Trie::deleteWord(const string& word) {
    // ----- 步骤1：检查是否存在 -----
    if (!wordSet_.count(word)) return false;

    // ----- 步骤2：定位到末节点 -----
    TrieNode* node = root_.get();
    for (char ch : word) {
        if (!node->children.count(ch)) return false;
        node = node->children[ch].get();
    }
    
    // ----- 步骤3：标记为非单词 -----
    node->isWord = false;
    node->freq = 0;
    node->word.clear();
    
    // ----- 步骤4：更新辅助索引 -----
    wordSet_.erase(word);
    wordFreq_.erase(word);

    // ----- 步骤5：缓存失效 -----
    invalidateCacheByString(word);

    return true;
}
```

### 图解：删除 "and"

```
删除前:                     删除后:
                            
    [root]                    [root]
     /                          /
    a                          a
     \                          \
      n                          n
       \                          \
        d                          d
        [isWord=true]             [isWord=false]  ← 标记为 false
        [freq=6]                  [freq=0]
        [word="and"]              [word=""]
```

**注意**：节点本身没有删除，只是标记为非单词。
- 好处：如果还有 "ando"，共享的 "and" 路径保留
- 代价：少量内存未释放

### 3. updateFreq - 更新词频

```cpp
bool Trie::updateFreq(const string& word, int freq) {
    // ----- 步骤1：参数校验 -----
    if (!wordSet_.count(word) || freq <= 0) return false;

    // ----- 步骤2：定位到末节点 -----
    TrieNode* node = root_.get();
    for (char ch : word) {
        if (!node->children.count(ch)) return false;
        node = node->children[ch].get();
    }
    
    // ----- 步骤3：更新词频 -----
    node->freq = freq;
    wordFreq_[word] = freq;

    // ----- 步骤4：缓存失效 -----
    invalidateCacheByString(word);

    return true;
}
```

## 五、查询函数详解

### 1. getFreq - 精确词频查询

```cpp
int Trie::getFreq(const string& word) const {
    auto it = wordFreq_.find(word);
    return (it != wordFreq_.end()) ? it->second : -1;
}
```

**时间复杂度**：O(1) (哈希表查找)

---

### 2. getAllWords - 全词库导出

```cpp
vector<pair<string, int>> Trie::getAllWords() const {
    vector<pair<string, int>> result;
    for (const auto& [word, freq] : wordFreq_)
        result.emplace_back(word, freq);
    sort(result.begin(), result.end(),
         [](const auto& a, const auto& b) { return a.first < b.first; });
    return result;
}
```

**时间复杂度**：O(N log N) (N 为单词总数，主要是排序开销)

### 3. prefixQuery - 前缀查询

```cpp
vector<pair<string, int>> Trie::prefixQuery(const string& prefix) const {
    vector<pair<string, int>> result;
    TrieNode* node = findNode(prefix);   // 1. 定位前缀节点
    if (node) {
        collectWords(node, result);      // 2. DFS 收集所有单词
        sortByFreqAndLex(result);        // 3. 排序
    }
    return result;
}
```

```
sortByFreqAndLex 排序后：
    and(6), ant(6), ando(4), another(3), an(2)
    ↑词频6    ↑词频6   ↑词频4    ↑词频3     ↑词频2
```

### 4. digitQuery - 数字序列查询（含缓存）

```cpp
vector<pair<string, int>> Trie::digitQuery(const string& digits) const {
    // 1. 缓存命中检查
    auto it = cache_.find(digits);
    if (it != cache_.end()) {
        return it->second;  // O(1) 直接返回
    }

    // 2. 缓存未命中，执行查询
    vector<pair<string, int>> result;
    collectByDigit(root_.get(), "", digits, 0, result);
    sortByFreqAndLex(result);
    if (result.size() > K) result.resize(K);  // Top-K 截断

    // 3. 存入缓存
    const_cast<Trie*>(this)->cache_[digits] = result;

    return result;
}
```

### 图解：digitQuery("26")

```
词库: an(2), and(6), ando(4), ant(6), another(3)

映射: 2→{a,b,c}, 6→{m,n,o}

DFS 匹配路径:
    '2' → 选择 'a'
    '6' → 选择 'n'
    到达 "an" 节点 → 收集所有单词

匹配结果:
    an(2), and(6), ando(4), ant(6), another(3)

排序后 Top-5:
    1. and(6), 2. ant(6), 3. ando(4), 4. another(3), 5. an(2)
```

### 缓存机制

```
第一次 input 26:
    cache_.find("26") → 未找到
    执行查询 → 排序 → 存入 cache_["26"]
    返回结果

第二次 input 26:
    cache_.find("26") → 找到 ✓
    直接返回缓存结果  (O(1) 极快)

add hello 10:
    invalidateCacheByString("hello")
    清除 "hello" 所有前缀的缓存
    cache_["26"] 被清空 (因为 'h' 影响 '4', '43', ...)
```

### 5. matchPattern - 通配符匹配

```cpp
vector<string> Trie::matchPattern(const string& pattern) const {
    vector<string> result;
    collectByPattern(root_.get(), "", pattern, 0, result);
    sort(result.begin(), result.end());
    return result;
}
```

### 通配符规则

| 符号 | 含义 | 示例 |
|------|------|------|
| `?` | 匹配任意单个字符 | `ba?d` → `band` |
| `*` | 匹配任意长度（含空） | `b*` → `banana`, `band`, `bar`, `bark` |

### 递归匹配逻辑

```cpp
void collectByPattern(node, current, pattern, pos, result) {
    if (pos == pattern.length()) {
        if (node->isWord) result.push_back(node->word);
        return;
    }
    
    char p = pattern[pos];
    
    if (p == '*') {
        // 情况A: * 匹配空串 → 跳过 *
        collectByPattern(node, current, pattern, pos + 1, result);
        
        // 情况B: * 匹配至少一个字符 → 吃掉当前字符，* 保留
        for (auto& [ch, child] : node->children)
            collectByPattern(child, current + ch, pattern, pos, result);
            
    } else if (p == '?') {
        // ? 匹配任意单个字符 → 遍历所有子节点
        for (auto& [ch, child] : node->children)
            collectByPattern(child, current + ch, pattern, pos + 1, result);
            
    } else {
        // 精确匹配
        if (node->children.count(p))
            collectByPattern(node->children[p], current + p, pattern, pos + 1, result);
    }
}
```

### 图解：matchPattern("b*")

```
词库: banana, band, bar, bark, car

        [root]
         /    \
        b      c
       /        \
      a          a
     /|\         r
    n r r
   /  |  \
  a   a   k
 /    |
n     k
|
a

匹配 "b*"：
    pos=0, 'b' 精确匹配 → 进入 'b' 节点
    pos=1, '*' 匹配空串 → 收集当前节点下所有单词
    pos=1, '*' 匹配至少一个字符 → 继续深入

结果（按字典序）：
    banana, band, bar, bark
```

## 六、缓存管理详解

### invalidateCacheByString - 前缀链失效

```cpp
void Trie::invalidateCacheByString(const string& str) {
    if (str.empty()) {
        invalidateAllCache();
        return;
    }

    // 收集所有前缀对应的数字序列
    vector<string> prefixDigits;
    collectAllPrefixDigits(str, prefixDigits);

    // 清空这些前缀的缓存
    for (const string& digits : prefixDigits) {
        invalidateCacheByDigits(digits);
    }

    // 清空该字符串本身的缓存
    string fullDigits = wordToDigits(str);
    if (!fullDigits.empty()) {
        invalidateCacheByDigits(fullDigits);
    }
}
```

### 图解：invalidateCacheByString("hello")

```
"hello" 映射:
    h → 4
    e → 3
    l → 5
    l → 5
    o → 6

collectAllPrefixDigits("hello"):
    第1轮: "h"  → "4"
    第2轮: "he" → "43"
    第3轮: "hel" → "435"
    第4轮: "hell" → "4355"
    第5轮: "hello" → "43556"

失效的缓存 Key:
    "4", "43", "435", "4355", "43556"

原理: 更新 "hello" 会影响所有以 'h' 开头的前缀查询
      "h" → 包含 hello
      "he" → 包含 hello
      "hel" → 包含 hello
      所以这些缓存都必须清空
```

### 为什么是"精细"缓存失效？

```
❌ 粗粒度: 修改任何单词 → 清空所有缓存 (cache_.clear())
   → 简单但效率低，大量缓存被无辜清除

✅ 细粒度: 修改 "hello" → 只清除 "4","43","435","4355","43556"
   → 复杂但效率高，不影响其他前缀的缓存 (如 "9"→"wxyz")
```


### collectByDigit - 数字匹配

```cpp
void Trie::collectByDigit(TrieNode* node, const string& current,
                          const string& digits, size_t pos,
                          vector<pair<string, int>>& result) const {
    if (pos == digits.length()) {
        collectWords(node, result);
        return;
    }
    char digit = digits[pos];
    auto it = KEY_MAP.find(digit);
    if (it == KEY_MAP.end()) return;
    for (char ch : it->second) {
        if (node->children.count(ch))
            collectByDigit(node->children[ch].get(), current + ch,
                           digits, pos + 1, result);
    }
}
```

### sortByFreqAndLex - 统一排序

```cpp
void Trie::sortByFreqAndLex(vector<pair<string, int>>& items) {
    sort(items.begin(), items.end(),
         [](const auto& a, const auto& b) {
             if (a.second != b.second) return a.second > b.second;  // 词频降序
             return a.first < b.first;                              // 字典序升序
         });
}
```

| 排序规则 | 代码 |
|----------|------|
| 词频降序 | `a.second > b.second` |
| 字典序升序 | `a.first < b.first` |

## 九、复杂度总结

| 操作 | 时间复杂度 | 空间复杂度 |
|------|------------|------------|
| `addWord` | O(L) | O(L) |
| `deleteWord` | O(L) | O(1) |
| `updateFreq` | O(L) | O(1) |
| `getFreq` | O(1) | O(1) |
| `getAllWords` | O(N log N) | O(N) |
| `prefixQuery` | O(L + M log M) | O(M) |
| `digitQuery` (未命中) | O(L·4^L + M log M) | O(M) |
| `digitQuery` (命中) | O(1) | O(1) |
| `matchPattern` | O(Σ^L) 最坏 | O(M) |
| `invalidateCacheByString` | O(L) | O(L) |

**变量说明**：
- L = 单词/模式长度
- N = 单词总数
- M = 匹配候选数
- K = Top-K 截断 (K=5)
- Σ = 字母表大小

```cpp
//   2. 检查序号是否有效
//   3. 选中对应单词，词频 +1
//   4. 候选列表失效
void NineKeySystem::cmdSelect(const vector<string>& args) {
    // ----- 1. 检查参数 -----
    if (args.size() != 2) {
        cout << "用法: select <序号>" << endl;
        return;
    }

    // ----- 2. 检查是否有候选列表 -----
    if (!hasCandidates_) {
        cout << "错误: 没有候选列表，请先执行 input" << endl;
        return;
    }

    // ----- 3. 解析序号 -----
    int rank;
    if (!toInt(args[1], rank) || rank <= 0) {
        cout << "错误: 序号必须为正整数" << endl;
        return;
    }

    // ----- 4. 检查序号是否越界 -----
    if ((size_t)rank > lastCandidates_.size()) {
        cout << "错误: 序号 " << rank << " 超出范围 (1-" 
             << lastCandidates_.size() << ")" << endl;
        return;
    }

    // ----- 5. 选择成功，词频 +1 -----
    const string& word = lastCandidates_[rank - 1].first;
    int newFreq = trie_.getFreq(word) + 1;
    trie_.updateFreq(word, newFreq);
    cout << "已选择 " << word << "，当前词频 = " << newFreq << endl;

    // ----- 6. 候选列表失效（任务书要求）-----
    invalidateCache();
}

        return;
    }

    auto r = trie_.prefixQuery(args[1]);
    if (r.empty()) {
        cout << "没有以 '" << args[1] << "' 开头的单词" << endl;
    } else {
        for (auto& [w, f] : r) {
            cout << w << " " << f << endl;
        }
    }
}

// ---------- 12.2 验证数字序列 ----------
//
// 规则：非空，且只包含字符 '2' 到 '9'
bool NineKeySystem::isValidDigits(const string& s) const {
    if (s.empty()) return false;
    for (char c : s) {
        if (c < '2' || c > '9') return false;
    }
    return true;
}

// ---------- 12.3 字符串转整数 ----------
//
// 使用 std::stoi 转换
// 检查是否整个字符串都被转换
bool NineKeySystem::toInt(const string& s, int& v) const {
    try {
        size_t p;
        v = stoi(s, &p);
        return p == s.size();  // 确保整个字符串都被转换
    } catch (...) {
        return false;  // 转换异常
    }
}

// ---------- 12.4 去除首尾空白 ----------
//
// 移除字符串首尾的空白字符（空格和制表符）
string NineKeySystem::trim(const string& s) const {
    size_t l = s.find_first_not_of(" \t");
    if (l == string::npos) return "";
    size_t r = s.find_last_not_of(" \t");
    return s.substr(l, r - l + 1);
}

// ---------- 12.5 分割命令 ----------
//
// 按空格分割字符串
// 例: "add and 6" → ["add", "and", "6"]
vector<string> NineKeySystem::split(const string& s) const {
    vector<string> r;
    istringstream iss(s);
    string t;
    while (iss >> t) {
        r.push_back(t);
    }
    return r;
}
```
