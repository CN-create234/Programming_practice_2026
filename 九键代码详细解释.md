```cpp
 * 作用：存放所有不变的配置数据，供多个文件共享

#ifndef CONSTANTS_H
#define CONSTANTS_H

#include <string>
#include <unordered_map>

using namespace std;

// ============================================================
// 1. Top-K 候选数量
// ============================================================
// 根据任务书要求，每次展示前 K = 5 个候选词
// 如果匹配的单词少于 5 个，则全部展示
const int K = 5;

// ============================================================
// 2. 正向映射：数字 → 字母（九键核心映射）
// ============================================================
// 用途：输入数字时，展开为所有可能的字母
// 例如：数字 '2' 对应字母 'a', 'b', 'c'
// 数据结构：每个数字对应一个字符串（包含3-4个字母）
// 
// 为什么用 unordered_map？
// → 哈希表，查找 O(1)，比数组更灵活
//
// 注意：7 和 9 有 4 个字母，其他有 3 个
const unordered_map<char, string> KEY_MAP = {
    {'2', "abc"},
    {'3', "def"},
    {'4', "ghi"},
    {'5', "jkl"},
    {'6', "mno"},
    {'7', "pqrs"},
    {'8', "tuv"},
    {'9', "wxyz"}
};

// ============================================================
// 3. 反向映射：字母 → 数字（用于缓存管理）
// ============================================================
// 用途：知道某个单词后，反向计算它的数字序列
// 例如：'h' → '4', 'e' → '3', 'l' → '5', 'l' → '5', 'o' → '6'
//       所以 "hello" → "43556"
//
// 为什么需要反向映射？

// → 例如更新 "hello" 的词频，需要清空 "43556" 的缓存
// → 没有反向映射，就无法从单词得到数字序列
const unordered_map<char, char> REVERSE_MAP = {
    {'a', '2'}, {'b', '2'}, {'c', '2'},

    {'j', '5'}, {'k', '5'}, {'l', '5'},
```

## 三、`trie.h` - Trie类声明

```cpp
/**
 * trie.h - Trie字典树类声明
 * 作用：定义数据结构和方法接口，类似于"目录"
 */

#ifndef TRIE_H
#define TRIE_H

#include <string>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <memory>
#include <utility>

using namespace std;

// ============================================================
// 1. TrieNode 结构体 - 字典树的节点
// ============================================================
// 
// 什么是 Trie（字典树）？
// ─────────────────────────────────────────────────────────────
// Trie 是一种树形数据结构，用于高效存储和查找字符串。
// 每个节点代表一个字符，从根到某个节点的路径形成一个前缀。
//
// 示例：存储 and, ando, another
//
//          root
//         /    \
//        a      t
//       / \      \
//      n   n      a
//     / \   \      \
//    d   o   o      n
//   /     \   \      \
//  and    ando anot   d
//  (6)    (4)  (3)    |
//                    tand
//                    (7)
//
// 每个节点存储：
//   - children: 子节点（当前字符的下一个字符）
//   - isWord:   是否是一个单词的结尾
//   - freq:     词频（仅当 isWord=true 时有效）
//   - word:     完整单词（仅当 isWord=true 时有效）
struct TrieNode {
    // 子节点映射
    // key: 字符 ('a' ~ 'z')
    // value: 指向子节点的智能指针
    // 
    // 为什么用 unique_ptr？
    // → 自动管理内存，不需要手动 delete
    // → 每个节点只能有一个父节点，所有权唯一
    // → 安全，不会内存泄漏
    unordered_map<char, unique_ptr<TrieNode>> children;

    // 是否是一个单词的结尾节点
    // true: 从根到该节点的路径构成一个完整单词
    // false: 只是一个前缀，不是完整单词
    bool isWord = false;

    // 词频：该单词被用户选择的累计次数
    // 仅当 isWord == true 时有效
    // 用于排序：词频越高越靠前
    int freq = 0;

    // 完整单词字符串
    // 仅当 isWord == true 时有效
    // 存储完整单词，方便直接读取
    string word;
};

// ============================================================
// 2. Trie 类 - 字典树的核心操作
// ============================================================
//
// 职责：
//   - 存储所有单词及其词频
//   - 提供添加、删除、更新词频功能
//   - 提供数字查询、前缀查询、通配符查询
//   - 管理 Top-K 缓存
class Trie {
public:
    // ---------- 构造函数 ----------
    // 初始化：创建根节点
    Trie();

    // ---------- 词库维护（增删改）----------
    // 添加单词
    // @param word: 要添加的单词（小写字母）
    // @param freq: 初始词频（正整数）
    // @return true: 添加成功，false: 失败（已存在或非法）
    bool addWord(const string& word, int freq);

    // 删除单词
    // @param word: 要删除的单词
    // @return true: 删除成功，false: 单词不存在
    bool deleteWord(const string& word);

    // 更新词频
    // @param word: 要更新的单词
    // @param freq: 新的词频（正整数）
    // @return true: 更新成功，false: 单词不存在
    bool updateFreq(const string& word, int freq);

    // ---------- 查询功能 ----------
    // 精确查询词频
    // @param word: 要查询的单词
    // @return 词频值，如果不存在返回 -1
    int getFreq(const string& word) const;

    // 全词库导出
    // @return 所有单词及词频，按字典序排序
    vector<pair<string, int>> getAllWords() const;

    // 前缀查询
    // @param prefix: 要查询的前缀
    // @return 所有以 prefix 开头的单词，按"词频降序+字典序升序"
    vector<pair<string, int>> prefixQuery(const string& prefix) const;

    // 数字序列查询（核心功能！）
    // @param digits: 数字序列（如 "26"）
    // @return 匹配该数字序列的 Top-K 个单词
    // 
    // 数字匹配规则：
    // 2→abc, 3→def, 4→ghi, 5→jkl,
    // 6→mno, 7→pqrs, 8→tuv, 9→wxyz
    vector<pair<string, int>> digitQuery(const string& digits) const;

    // 通配符匹配
    // @param pattern: 匹配模式
    //   - '?' 匹配任意单个字符
    //   - '*' 匹配任意长度字符串（包括空串）
    // @return 所有匹配的单词，按字典序排序
    vector<string> matchPattern(const string& pattern) const;

    // ---------- 清空 ----------
    // 清空所有数据
    void clear();

    // ---------- 缓存管理 ----------
    // 清空所有缓存
    void invalidateAllCache();

    // 清空指定数字序列的缓存
    void invalidateCacheByDigits(const string& digits);

    // 清空单词相关的所有缓存（影响所有前缀）
    void invalidateCacheByWord(const string& word);

    // 清空前缀相关的缓存
    void invalidateCacheByPrefix(const string& prefix);

    // 打印缓存统计（调试用）
    void printCacheStats() const;

private:
    // ============================================================
    // 3. 私有成员变量
    // ============================================================

    // 根节点（树的入口）
    // unique_ptr 自动管理根节点的生命周期
    unique_ptr<TrieNode> root_;

    // 单词集合（快速判断单词是否存在）
    // 哈希集合，查找 O(1)
    // 用于 add/delete/update 时快速检查单词是否存在
    unordered_set<string> wordSet_;

    // 词频映射表（快速查询词频）
    // 哈希表，查找 O(1)
    // 用于 getFreq 时快速返回结果
    // 也用于 getAllWords 时遍历所有单词
    unordered_map<string, int> wordFreq_;

    // ===== Top-K 缓存 =====
    // key: 数字序列 (如 "26")
    // value: 该数字序列对应的 Top-K 候选列表 (已排序)
    // 
    // 为什么要缓存？
    // → 用户可能多次输入相同的数字序列
    // → 每次从 Trie 查询都要遍历和排序，耗时
    // → 缓存可以 O(1) 返回结果
    //
    // 什么时候更新缓存？
    // → 任何词库修改（add/delete/update）都需要清空相关缓存
    // → 选择单词后词频变化，也需要清空
    unordered_map<string, vector<pair<string, int>>> cache_;

    // ============================================================
    // 4. 私有辅助函数
    // ============================================================

    // 校验单词是否合法（非空、全小写字母）
    bool isValidWord(const string& word) const;

    // 根据前缀查找对应的节点
    // @param prefix: 要查找的前缀
    // @return 指向该节点的指针，不存在返回 nullptr
    TrieNode* findNode(const string& prefix) const;

    // DFS 收集从当前节点开始的所有完整单词
    // @param node: 当前节点
    // @param result: 收集结果的引用
    void collectWords(TrieNode* node, vector<pair<string, int>>& result) const;

    // 数字匹配的递归实现
    // @param node: 当前节点
    // @param current: 当前已匹配的字符串
    // @param digits: 完整的数字序列
    // @param pos: 当前处理的数字位置
    // @param result: 收集结果的引用
    void collectByDigit(TrieNode* node, const string& current,
                        const string& digits, size_t pos,
                        vector<pair<string, int>>& result) const;

    // 通配符匹配的递归实现
    void collectByPattern(TrieNode* node, const string& current,
                          const string& pattern, size_t pos,
                          vector<string>& result) const;

    // 排序：词频降序，字典序升序
    // 任务书要求的统一排序规则
    static void sortByFreqAndLex(vector<pair<string, int>>& items);

    // ===== 缓存辅助函数 =====
    // 单词转数字序列
    // 例: "hello" → "43556"
    string wordToDigits(const string& word) const;

    // 前缀转数字序列
    // 例: "hel" → "435"
    string prefixToDigits(const string& prefix) const;

    // 收集单词的所有前缀对应的数字序列
    // 例: "hello" → ["4", "43", "435", "4355", "43556"]
    void collectAllPrefixDigits(const string& word,
                                vector<string>& prefixes) const;

    // 从前缀开始收集所有前缀对应的数字序列
    void collectAllPrefixDigitsFromPrefix(const string& prefix,
                                          vector<string>& prefixes) const;
};

#endif
```

## 四、`trie.cpp` - Trie类实现

```cpp
/**
 * trie.cpp - Trie字典树类实现
 * 这是整个系统的核心数据层
 */

#include "trie.h"
#include "constants.h"
#include <algorithm>
#include <cctype>
#include <iostream>

using namespace std;

// ============================================================
// 1. 构造函数
// ============================================================
// 创建根节点，根节点不存储任何字符
Trie::Trie() : root_(make_unique<TrieNode>()) {}

// ============================================================
// 2. 添加单词
// ============================================================
//
// 算法流程：
// ┌─────────────────────────────────────────────────────────────┐
// │  用户: add and 6                                          │
// │                                                           │
// │  步骤1: 检查合法性                                        │
// │    - 单词非空且全小写 ✓                                   │
// │    - 词频 > 0 ✓                                          │
// │    - 单词不存在 ✓                                        │
// │                                                           │
// │  步骤2: 从根节点开始，逐字符创建路径                      │
// │    root ──a──► node_a ──n──► node_an ──d──► node_and   │
// │                                                           │
// │  步骤3: 在末尾节点标记单词                                │
// │    node_and.isWord = true                                │
// │    node_and.freq = 6                                     │
// │    node_and.word = "and"                                 │
// │                                                           │
// │  步骤4: 更新辅助数据结构                                  │
// │    wordSet_.insert("and")                                │
// │    wordFreq_["and"] = 6                                  │
// │                                                           │
// │  步骤5: 清空相关缓存                                      │
// │    invalidateCacheByWord("and")                          │
// │    清空 "2", "26", "263" 等前缀的缓存                    │
// └─────────────────────────────────────────────────────────────┘
bool Trie::addWord(const string& word, int freq) {
    // ----- 1. 合法性检查 -----
    // isValidWord: 检查是否非空且全小写字母
    // freq <= 0: 词频必须为正整数
    // wordSet_.count(word): 检查单词是否已存在
    if (!isValidWord(word) || freq <= 0 || wordSet_.count(word))
        return false;  // 添加失败

    // ----- 2. 从根节点开始遍历 -----
    TrieNode* node = root_.get();  // 从根节点开始
    for (char ch : word) {
        // 如果当前字符没有对应的子节点，创建一个新节点
        if (!node->children.count(ch)) {
            // make_unique<TrieNode>() 创建新节点
            // 使用 unique_ptr 自动管理内存
            node->children[ch] = make_unique<TrieNode>();
        }
        // 移动到子节点
        node = node->children[ch].get();
    }

    // ----- 3. 到达单词末尾，标记为单词节点 -----
    node->isWord = true;   // 标记这是一个完整单词
    node->freq = freq;     // 存储词频
    node->word = word;     // 存储完整单词

    // ----- 4. 更新辅助数据结构 -----
    wordSet_.insert(word);          // 加入集合，用于快速判断存在性
    wordFreq_[word] = freq;         // 加入映射表，用于快速查询词频

    // ----- 5. 精细缓存更新 -----
    // 新词会影响其所有前缀的查询结果
    // 例如添加 "hello"，会影响 "h", "he", "hel", "hell", "hello" 的缓存
    invalidateCacheByWord(word);

    return true;  // 添加成功
}

// ============================================================
// 3. 删除单词
// ============================================================
//
// 算法流程：
// ┌─────────────────────────────────────────────────────────────┐
// │  用户: delete and                                         │
// │                                                           │
// │  步骤1: 检查单词是否存在                                  │
// │                                                           │
// │  步骤2: 找到单词末尾节点                                  │
// │    root → a → n → d                                      │
// │                                                           │
// │  步骤3: 取消标记（不删除节点！）                          │
// │    node.isWord = false                                   │
// │    node.freq = 0                                         │
// │    node.word.clear()                                     │
// │                                                           │
// │  步骤4: 更新辅助数据结构                                  │
// │    wordSet_.erase("and")                                 │
// │    wordFreq_.erase("and")                                │
// │                                                           │
// │  为什么只取消标记不删除节点？                             │
// │    → 共享前缀的其他单词可能还在                          │
// │    → 例如 "ando" 共享 "and" 路径                         │
// │    → 删除节点会破坏 "ando" 的路径                        │
// └─────────────────────────────────────────────────────────────┘
bool Trie::deleteWord(const string& word) {
    // ----- 1. 检查单词是否存在 -----
    if (!wordSet_.count(word)) return false;

    // ----- 2. 找到单词末尾节点 -----
    TrieNode* node = root_.get();
    for (char ch : word) {
        if (!node->children.count(ch)) return false;  // 路径不存在
        node = node->children[ch].get();
    }

    // ----- 3. 取消单词标记（保留节点）-----
    node->isWord = false;   // 不再是完整单词
    node->freq = 0;         // 清空词频
    node->word.clear();     // 清空存储的单词

    // ----- 4. 更新辅助数据结构 -----
    wordSet_.erase(word);
    wordFreq_.erase(word);

    // ----- 5. 精细缓存更新 -----
    invalidateCacheByWord(word);

    return true;
}

// ============================================================
// 4. 更新词频
// ============================================================
//
// 算法流程：
// ┌─────────────────────────────────────────────────────────────┐
// │  用户: update and 10                                      │
// │                                                           │
// │  步骤1: 检查单词是否存在且词频合法                        │
// │                                                           │
// │  步骤2: 找到单词末尾节点                                  │
// │                                                           │
// │  步骤3: 更新节点中的词频                                  │
// │    node.freq = 10                                        │
// │    wordFreq_["and"] = 10                                 │
// │                                                           │
// │  步骤4: 清空相关缓存                                      │
// │    词频变化影响排序，所有前缀缓存都要清空                │
// └─────────────────────────────────────────────────────────────┘
bool Trie::updateFreq(const string& word, int freq) {
    // ----- 1. 检查合法性 -----
    if (!wordSet_.count(word) || freq <= 0) return false;

    // ----- 2. 找到单词末尾节点 -----
    TrieNode* node = root_.get();
    for (char ch : word) {
        if (!node->children.count(ch)) return false;
        node = node->children[ch].get();
    }

    // ----- 3. 更新词频 -----
    node->freq = freq;
    wordFreq_[word] = freq;

    // ----- 4. 精细缓存更新 -----
    invalidateCacheByWord(word);

    return true;
}

// ============================================================
// 5. 查询词频
// ============================================================
// 直接使用哈希表查找，O(1) 时间复杂度
int Trie::getFreq(const string& word) const {
    auto it = wordFreq_.find(word);
    // 如果找到返回词频，否则返回 -1
    return (it != wordFreq_.end()) ? it->second : -1;
}

// ============================================================
// 6. 全词库导出
// ============================================================
//
// 算法流程：
// ┌─────────────────────────────────────────────────────────────┐
// │  1. 从 wordFreq_ 哈希表中收集所有单词                     │
// │  2. 按字典序排序（任务书要求）                            │
// │  3. 返回结果                                              │
// └─────────────────────────────────────────────────────────────┘
vector<pair<string, int>> Trie::getAllWords() const {
    // ----- 1. 收集所有单词 -----
    vector<pair<string, int>> result;
    for (const auto& [word, freq] : wordFreq_) {
        result.emplace_back(word, freq);
    }

    // ----- 2. 按字典序排序 -----
    // 使用 lambda 表达式作为比较函数
    // a.first < b.first 表示按单词字符串升序排列
    sort(result.begin(), result.end(),
         [](const auto& a, const auto& b) { return a.first < b.first; });

    return result;
}

// ============================================================
// 7. 前缀查询
// ============================================================
//
// 算法流程：
// ┌─────────────────────────────────────────────────────────────┐
// │  用户: prefix an                                          │
// │                                                           │
// │  步骤1: 找到前缀 "an" 对应的节点                         │
// │    root → a → n                                          │
// │                                                           │
// │  步骤2: 从该节点开始，DFS 收集所有完整单词                │
// │    an 节点下有两个分支:                                   │
// │    - d → and(6)                                          │
// │    - o → ando(4)                                         │
// │    - o → another(3) (在另一个分支)                       │
// │                                                           │
// │  步骤3: 按"词频降序+字典序升序"排序                       │
// │    and(6), ando(4), another(3)                           │
// └─────────────────────────────────────────────────────────────┘
vector<pair<string, int>> Trie::prefixQuery(const string& prefix) const {
    vector<pair<string, int>> result;

    // ----- 1. 找到前缀对应的节点 -----
    TrieNode* node = findNode(prefix);
    if (node) {
        // ----- 2. 从该节点收集所有单词 -----
        collectWords(node, result);

        // ----- 3. 按统一规则排序 -----
        sortByFreqAndLex(result);
    }

    return result;
}

// ============================================================
// 8. 数字序列查询（核心功能！带缓存）
// ============================================================
//
// 算法流程：
// ┌─────────────────────────────────────────────────────────────┐
// │  用户: input 26                                           │
// │                                                           │
// │  步骤1: 检查缓存                                          │
// │    cache_.find("26") → 如果存在直接返回                   │
// │                                                           │
// │  步骤2: 缓存未命中，执行查询                              │
// │    collectByDigit(root, "", "26", 0)                     │
// │    数字 '2' → 尝试 'a', 'b', 'c'                         │
// │    数字 '6' → 尝试 'm', 'n', 'o'                         │
// │    匹配结果: and(6), ando(4), another(3)                 │
// │                                                           │
// │  步骤3: 排序并取 Top-K                                    │
// │    sortByFreqAndLex() → 按词频降序+字典序升序            │
// │    resize(K) → 只取前5个                                 │
// │                                                           │
// │  步骤4: 存入缓存                                          │
// │    cache_["26"] = result                                 │
// └─────────────────────────────────────────────────────────────┘
vector<pair<string, int>> Trie::digitQuery(const string& digits) const {
    // ----- 1. 检查缓存 -----
    auto it = cache_.find(digits);
    if (it != cache_.end()) {
        return it->second;  // 缓存命中，直接返回
    }

    // ----- 2. 缓存未命中，执行查询 -----
    vector<pair<string, int>> result;
    collectByDigit(root_.get(), "", digits, 0, result);

    // ----- 3. 排序并取 Top-K -----
    sortByFreqAndLex(result);
    if (result.size() > K) {
        result.resize(K);  // 只保留前 K 个
    }

    // ----- 4. 存入缓存 -----
    // const_cast 去掉 const 限制，以便修改 cache_
    // 因为这是 const 方法，但缓存修改不影响逻辑状态
    const_cast<Trie*>(this)->cache_[digits] = result;

    return result;
}

// ============================================================
// 9. 通配符匹配
// ============================================================
//
// 通配符规则：
//   - '?' 匹配任意单个字符
//   - '*' 匹配任意长度字符串（包括空字符串）
//
// 算法流程：
// ┌─────────────────────────────────────────────────────────────┐
// │  用户: match a*d                                          │
// │                                                           │
// │  递归展开:                                                │
// │  match(root, "", "a*d", 0)                               │
// │    ├─ 字符 'a' → 必须匹配 'a'                            │
// │    │     └─ match(node_a, "a", "*d", 1)                  │
// │    │           ├─ '*' 匹配空串 → match(node_a, "a", "d", 2)│
// │    │           │     └─ 必须匹配 'd' → 找到 and          │
// │    │           └─ '*' 匹配一个字符 → 继续匹配 '*'         │
// │    │                 ├─ 匹配 'n' → match(node_an, "an", "*d", 1)│
// │    │                 │     ├─ '*' 匹配空串 → 需要 'd' → and│
// │    │                 │     └─ 继续匹配...                 │
// │    │                 └─ ...                              │
// │    └─ 结果: and, abcd, around, ...                       │
// └─────────────────────────────────────────────────────────────┘
vector<string> Trie::matchPattern(const string& pattern) const {
    vector<string> result;
    collectByPattern(root_.get(), "", pattern, 0, result);

    // 按字典序排序（任务书要求）
    sort(result.begin(), result.end());

    return result;
}

// 10. 清空
// ============================================================
void Trie::clear() {
    root_ = make_unique<TrieNode>();  // 重置根节点
    wordSet_.clear();                  // 清空单词集合
    wordFreq_.clear();                // 清空词频映射
    cache_.clear();                   // 清空缓存
}

// ============================================================
// 11. 精细缓存管理
// ============================================================

// 清空所有缓存
void Trie::invalidateAllCache() {
    cache_.clear();
}

// 清空指定数字序列的缓存
void Trie::invalidateCacheByDigits(const string& digits) {
    auto it = cache_.find(digits);
    if (it != cache_.end()) {
        cache_.erase(it);
    }
}

// 清空单词相关的所有缓存
// 原理：单词影响其所有前缀的查询结果
// 例：更新 "hello" → 清空 "4", "43", "435", "4355", "43556" 的缓存
void Trie::invalidateCacheByWord(const string& word) {
    if (word.empty()) {
        invalidateAllCache();
        return;
    }

    // ----- 1. 收集所有前缀的数字序列 -----
    vector<string> prefixDigits;
    collectAllPrefixDigits(word, prefixDigits);

    // ----- 2. 逐个清空这些前缀的缓存 -----
    for (const string& digits : prefixDigits) {
        invalidateCacheByDigits(digits);
    }

    // ----- 3. 也清空完整单词的缓存 -----
    string fullDigits = wordToDigits(word);
    if (!fullDigits.empty()) {
        invalidateCacheByDigits(fullDigits);
    }
}

// 清空前缀相关的缓存
void Trie::invalidateCacheByPrefix(const string& prefix) {
    if (prefix.empty()) {
        invalidateAllCache();
        return;
    }

    vector<string> prefixDigits;
    collectAllPrefixDigitsFromPrefix(prefix, prefixDigits);

    for (const string& digits : prefixDigits) {
        invalidateCacheByDigits(digits);
    }

    string fullDigits = prefixToDigits(prefix);
    if (!fullDigits.empty()) {
        invalidateCacheByDigits(fullDigits);
    }
}

// 打印缓存统计（调试用）
void Trie::printCacheStats() const {
    cout << "===== 缓存统计 =====" << endl;
    cout << "缓存条目数: " << cache_.size() << endl;
    for (const auto& [digits, candidates] : cache_) {
        cout << "  " << digits << " -> " << candidates.size() << " 个候选" << endl;
    }
    cout << "===================" << endl;
}

// ---------- 12.1 校验单词是否合法 ----------
// 条件：非空、全小写字母
bool Trie::isValidWord(const string& word) const {
    if (word.empty()) return false;
    for (char ch : word) {
        // islower: 是否小写字母
        // isalpha: 是否字母
        if (!islower(ch) || !isalpha(ch)) return false;
    }
    return true;
}

// ---------- 12.2 根据前缀查找节点 ----------
TrieNode* Trie::findNode(const string& prefix) const {
    TrieNode* node = root_.get();
    for (char ch : prefix) {
        if (!node->children.count(ch)) {
            return nullptr;  // 前缀不存在
        }
        node = node->children[ch].get();
    }
    return node;
}

// ---------- 12.3 DFS 收集所有单词 ----------
// 
// 递归过程示例：
// collectWords(node_and, result)
//   → node_and.isWord = true → 添加 and
//   → 遍历子节点: 'o' → collectWords(node_ando, result)
//     → node_ando.isWord = true → 添加 ando
//     → 没有子节点，返回
//   → 返回
void Trie::collectWords(TrieNode* node, vector<pair<string, int>>& result) const {
    // 如果当前节点是单词，加入结果
    if (node->isWord) {
        result.emplace_back(node->word, node->freq);
    }

    // 递归遍历所有子节点
    for (auto& [ch, child] : node->children) {
        collectWords(child.get(), result);
    }
}

// ---------- 12.4 数字匹配递归 ----------
//
// 递归过程示例：digits = "26"
// collectByDigit(root, "", "26", 0)
//   → digit='2', letters="abc"
//   → 遍历 'a', 'b', 'c'
//   → 对 'a': collectByDigit(node_a, "a", "26", 1)
//       → digit='6', letters="mno"
//       → 遍历 'm', 'n', 'o'
//       → 对 'n': collectByDigit(node_an, "an", "26", 2)
//           → pos==length，调用 collectWords(node_an, result)
//           → 收集 and(6), ando(4), another(3)
void Trie::collectByDigit(TrieNode* node, const string& current,
                          const string& digits, size_t pos,
                          vector<pair<string, int>>& result) const {
    // 如果处理完所有数字，收集从当前节点开始的所有单词
    if (pos == digits.length()) {
        collectWords(node, result);
        return;
    }

    char digit = digits[pos];
    auto it = KEY_MAP.find(digit);
    if (it == KEY_MAP.end()) return;  // 非法数字

    const string& letters = it->second;
    // 遍历每个可能的字母
    for (char ch : letters) {
        if (node->children.count(ch)) {
            collectByDigit(node->children[ch].get(), current + ch,
                           digits, pos + 1, result);
        }
    }
}

// ---------- 12.5 通配符匹配递归 ----------
//
// 关键理解：
// - '*' 有两种处理方式：
//   1. 匹配空字符串 → 跳过 '*' 继续匹配后续模式
//   2. 匹配一个字符 → 消费一个字符，但 '*' 保留继续匹配
void Trie::collectByPattern(TrieNode* node, const string& current,
                            const string& pattern, size_t pos,
                            vector<string>& result) const {
    // 如果处理完所有模式字符
    if (pos == pattern.length()) {
        if (node->isWord) {
            result.push_back(node->word);
        }
        return;
    }

    char p = pattern[pos];

    if (p == '*') {
        // 情况1: '*' 匹配空字符串，跳过当前 '*' 字符
        collectByPattern(node, current, pattern, pos + 1, result);

        // 情况2: '*' 匹配至少一个字符
        // 遍历所有子节点，'*' 继续匹配（pos 不变）
        for (auto& [ch, child] : node->children) {
            collectByPattern(child.get(), current + ch, pattern, pos, result);
        }
    }
    else if (p == '?') {
        // '?' 匹配任意单个字符
        // 遍历当前节点的所有子节点
        // 每个子节点都代表一个可能的字符
        // pos + 1 表示消耗掉 '?'，继续匹配下一个模式字符
        for (auto& [ch, child] : node->children) {
            collectByPattern(child.get(), current + ch,
                             pattern, pos + 1, result);
        }
    }
    else {
        // 普通字符：精确匹配
        // 检查当前节点是否有该字符对应的子节点
        if (node->children.count(p)) {
            // 如果有，继续递归匹配
            collectByPattern(node->children[p].get(), current + p,
                             pattern, pos + 1, result);
        }
        // 如果没有，这条路径匹配失败，直接返回
    }
}

// ---------- 12.6 排序：词频降序，字典序升序 ----------
//
// 任务书要求的统一排序规则：
// 1. 第一关键字：词频从高到低
// 2. 第二关键字：词频相同则按字典序从小到大
void Trie::sortByFreqAndLex(vector<pair<string, int>>& items) {
    sort(items.begin(), items.end(),
         [](const auto& a, const auto& b) {
             // 先比较词频（降序）
             if (a.second != b.second) {
                 return a.second > b.second;  // 词频高的在前
             }
             // 词频相同，比较单词（字典序升序）
             return a.first < b.first;
         });
}

// ============================================================
// 13. 缓存辅助函数实现
// ============================================================

// ---------- 13.1 单词转数字序列 ----------
//
// 原理：利用反向映射表 REVERSE_MAP
// 每个字母 → 对应的数字
// 例: "hello" → "43556"
string Trie::wordToDigits(const string& word) const {
    string result;
    for (char ch : word) {
        auto it = REVERSE_MAP.find(ch);
        if (it != REVERSE_MAP.end()) {
            result += it->second;  // 拼接数字
        } else {
            return "";  // 非法字符（理论上不会发生）
        }
    }
    return result;
}

// ---------- 13.2 前缀转数字序列 ----------
//
// 与 wordToDigits 类似，但用于前缀
// 例: "hel" → "435"
string Trie::prefixToDigits(const string& prefix) const {
    string result;
    for (char ch : prefix) {
        auto it = REVERSE_MAP.find(ch);
        if (it != REVERSE_MAP.end()) {
            result += it->second;
        } else {
            return "";
        }
    }
    return result;
}

// ---------- 13.3 收集所有前缀的数字序列 ----------
//
// 原理：逐步累积，每次加入一个字母就记录一次
// 例: "hello"
//   第1步: "h" → "4"     → prefixes = ["4"]
//   第2步: "he" → "43"    → prefixes = ["4", "43"]
//   第3步: "hel" → "435"   → prefixes = ["4", "43", "435"]
//   第4步: "hell" → "4355"  → prefixes = ["4", "43", "435", "4355"]
//   第5步: "hello" → "43556" → prefixes = ["4", "43", "435", "4355", "43556"]
void Trie::collectAllPrefixDigits(const string& word,
                                  vector<string>& prefixes) const {
    string currentDigits;
    for (char ch : word) {
        auto it = REVERSE_MAP.find(ch);
        if (it != REVERSE_MAP.end()) {
            currentDigits += it->second;
            prefixes.push_back(currentDigits);
        }
    }
}

// ---------- 13.4 从前缀开始收集所有前缀的数字序列 ----------
//
// 与 collectAllPrefixDigits 类似，但用于任意前缀
// 例: prefix = "hello"
//   结果同样: ["4", "43", "435", "4355", "43556"]
void Trie::collectAllPrefixDigitsFromPrefix(const string& prefix,
                                            vector<string>& prefixes) const {
    string currentDigits;
    for (char ch : prefix) {
        auto it = REVERSE_MAP.find(ch);
        if (it != REVERSE_MAP.end()) {
            currentDigits += it->second;
            prefixes.push_back(currentDigits);
        }
    }
}
```

```cpp
// 格式: update <word> <freq>
// 示例: update and 10
void NineKeySystem::cmdUpdate(const vector<string>& args) {
    if (args.size() != 3) {
        cout << "用法: update <单词> <词频>" << endl;
        return;
    }

    int freq;
    if (!toInt(args[2], freq) || freq <= 0) {
        cout << "错误: 词频必须为正整数" << endl;
        return;
    }

    if (trie_.updateFreq(args[1], freq)) {
        cout << "成功更新: " << args[1] << " -> " << freq << endl;
        invalidateCache();
    } else {
        cout << "错误: 单词 '" << args[1] << "' 不存在" << endl;
    }
}

// ============================================================
// 5. 数字输入命令（核心！）
// ============================================================
//
// 格式: input <digits>
// 示例: input 26
//
// 功能：
//   1. 验证数字序列（只含 2-9）
//   2. 调用 Trie 查询
//   3. 保存候选列表供 select 使用
//   4. 显示结果
void NineKeySystem::cmdInput(const vector<string>& args) {
    // ----- 1. 检查参数 -----
    if (args.size() != 2) {
        cout << "用法: input <数字序列>" << endl;
        return;
    }

    // ----- 2. 验证数字序列 -----
    if (!isValidDigits(args[1])) {
        cout << "错误: 数字序列只能包含 2-9" << endl;
        invalidateCache();
        return;
    }

    // ----- 3. 查询候选词 -----
    lastCandidates_ = trie_.digitQuery(args[1]);
    hasCandidates_ = !lastCandidates_.empty();

    // ----- 4. 显示结果 -----
    if (lastCandidates_.empty()) {
        cout << "无匹配单词" << endl;
    } else {
        for (size_t i = 0; i < lastCandidates_.size(); ++i) {
            cout << (i + 1) << ". " << lastCandidates_[i].first
                 << " " << lastCandidates_[i].second << endl;
        }
    }
}

// ============================================================
// 6. 选择候选词命令
// ============================================================
//
// 格式: select <rank>
// 示例: select 2
//
// 功能：
//   1. 检查是否有有效的候选列表
//   2. 检查序号是否有效
//   3. 选中对应单词，词频 +1
//   4. 候选列表失效
void NineKeySystem::cmdSelect(const vector<string>& args) {
    // ----- 1. 检查参数 -----
    if (args.size() != 2) {
        cout << "用法: select <序号>" << endl;
        return;
    }

    // ----- 2. 检查是否有候选列表 -----
    if (!hasCandidates_) {
        cout << "错误: 没有候选列表，请先执行 input" << endl;
        return;
    }

    // ----- 3. 解析序号 -----
    int rank;
    if (!toInt(args[1], rank) || rank <= 0) {
        cout << "错误: 序号必须为正整数" << endl;
        return;
    }

    // ----- 4. 检查序号是否越界 -----
    if ((size_t)rank > lastCandidates_.size()) {
        cout << "错误: 序号 " << rank << " 超出范围 (1-" 
             << lastCandidates_.size() << ")" << endl;
        return;
    }

    // ----- 5. 选择成功，词频 +1 -----
    const string& word = lastCandidates_[rank - 1].first;
    int newFreq = trie_.getFreq(word) + 1;
    trie_.updateFreq(word, newFreq);
    cout << "已选择 " << word << "，当前词频 = " << newFreq << endl;

    // ----- 6. 候选列表失效（任务书要求）-----
    invalidateCache();
}

// 12. 辅助函数实现
// ============================================================

// ---------- 12.1 使候选列表失效 ----------
//
// 作用：
//   - 清空 lastCandidates_
//   - 设置 hasCandidates_ = false
//   - 后续 select 命令会报错
void NineKeySystem::invalidateCache() {
    hasCandidates_ = false;
    lastCandidates_.clear();
}

// ---------- 12.2 验证数字序列 ----------
//
// 规则：非空，且只包含字符 '2' 到 '9'
bool NineKeySystem::isValidDigits(const string& s) const {
    if (s.empty()) return false;
    for (char c : s) {
        if (c < '2' || c > '9') return false;
    }
    return true;
}

// ---------- 12.3 字符串转整数 ----------
//
// 使用 std::stoi 转换
// 检查是否整个字符串都被转换
bool NineKeySystem::toInt(const string& s, int& v) const {
    try {
        size_t p;
        v = stoi(s, &p);
        return p == s.size();  // 确保整个字符串都被转换
    } catch (...) {
        return false;  // 转换异常
    }
}

// ---------- 12.4 去除首尾空白 ----------
//
// 移除字符串首尾的空白字符（空格和制表符）
string NineKeySystem::trim(const string& s) const {
    size_t l = s.find_first_not_of(" \t");
    if (l == string::npos) return "";
    size_t r = s.find_last_not_of(" \t");
    return s.substr(l, r - l + 1);
}

// ---------- 12.5 分割命令 ----------
//
// 按空格分割字符串
// 例: "add and 6" → ["add", "and", "6"]
vector<string> NineKeySystem::split(const string& s) const {
    vector<string> r;
    istringstream iss(s);
    string t;
    while (iss >> t) {
        r.push_back(t);
    }
    return r;
}
```
